class Solution
{
    public:
    //Function to find a continuous sub-array which adds up to a given number.
    vector<int> subarraySum(int a[], int n, long long s)
    {
        long long t=0,l=0;
        for(int i=0;i<n;i++){
            t+=a[i];
            while(t>s&&l<i)t-=a[l++];
            if(t==s){
                return {l+1,i+1};
            }
        }
        return {-1};
    }
};
